\makeproblem{Convolution theorem.}{pr:convolution:1}{
Find the form of the Fourier transform
\begin{equation}\label{eqn:convolution:10}
\tilde{f}(\omega) = \int f(y) e^{-i \omega y} dy.
\end{equation}
%
of a convolution integral
%
\begin{equation}\label{eqn:convolution:30}
f(y) = \int dy' g(y') h(y - y') = g(y) \conj h(y).
\end{equation}
%
} % makeproblem

\makeanswer{pr:convolution:1}{
This is basically just an application of change of variables
%
\begin{equation}\label{eqn:convolution:50}
\begin{aligned}
\calF f(y)
&= \int dy e^{-i \omega y} dy' g(y') h(y - y') \\
&= \int dy' g(y') \int dy e^{-i \omega y} \mathLabelBox{h(y - y')}{ \(y - y' = u\) } \\
&= \int dy' g(y') \int du e^{-i \omega (u + y')} h(u) \\
&= \tilde{h}(\omega) \int dy' g(y') e^{-i \omega y'} \\
&= \tilde{h}(\omega) \tilde{g}(\omega).
\end{aligned}
\end{equation}
%
} % makeanswer
