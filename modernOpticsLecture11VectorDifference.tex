\makeproblem
{Spatial distribution vector difference.}
%{Perform the spatial distribution vector difference calculation}
{lecture11VectorDifference:pr:1}{
Referring to \cref{fig:modernOpticsLecture11:modernOpticsLecture11Fig4} calculate \(R_1 - R_2\).
} % makeproblem

\makeanswer{lecture11VectorDifference:pr:1}{
From the figure we see that we have
\begin{subequations}
\begin{dmath}\label{eqn:lecture11VectorDifference:10}
\Br_s + \BR_1 + \inv{2} \Delta \Br = \Br_{\mathrm{av}}
\end{dmath}
\begin{dmath}\label{eqn:lecture11VectorDifference:30}
\Br_s + \BR_2 - \inv{2} \Delta \Br = \Br_{\mathrm{av}},
\end{dmath}
\end{subequations}
or
\begin{subequations}
\begin{dmath}\label{eqn:lecture11VectorDifference:50}
\BR_1 = \Br_{\mathrm{av}} - \Br_s - \inv{2} \Delta \Br
\end{dmath}
\begin{dmath}\label{eqn:lecture11VectorDifference:70}
\BR_2 = \Br_{\mathrm{av}} - \Br_s + \inv{2} \Delta \Br.
\end{dmath}
\end{subequations}

Squaring for the magnitudes, we have
\begin{subequations}
\begin{dmath}\label{eqn:lecture11VectorDifference:90}
\BR_1^2 = \left( \Br_{\mathrm{av}} - \Br_s \right)^2 + \left(\inv{2} \Delta \Br \right)^2
- \Delta \Br \cdot \left(\Br_{\mathrm{av}} - \Br_s\right)
\end{dmath}
\begin{dmath}\label{eqn:lecture11VectorDifference:110}
\BR_2^2 = \left( \Br_{\mathrm{av}} - \Br_s \right)^2 + \left(\inv{2} \Delta \Br \right)^2
+ \Delta \Br \cdot \left(\Br_{\mathrm{av}} - \Br_s\right) .
\end{dmath}
\end{subequations}

Assuming that \(\Abs{\Br_{\mathrm{av}}} \gg \Abs{\Br_s}\), so that \(\Abs{\Br_{\mathrm{av}} - \Br_s} \sim \Abs{\Br_{\mathrm{av}}}\), we have
%
\begin{dmath}\label{eqn:lecture11VectorDifference:130}
R_{1,2}^2 \sim
r_{\mathrm{av}}^2
\left(
1
+ \left(\inv{2 r_{\mathrm{av}}} \Delta \Br \right)^2
\mp \inv{r_{\mathrm{av}}^2} \Delta \Br \cdot \left(\Br_{\mathrm{av}} - \Br_s\right) .
\right),
\end{dmath}

which, to first order, is
%
\begin{dmath}\label{eqn:lecture11VectorDifference:150}
R_{1,2} \sim
r_{\mathrm{av}}
+ \inv{2 r_{\mathrm{av}}} \left(\inv{2} \Delta \Br \right)^2
\mp \inv{2 r_{\mathrm{av}}} \Delta \Br \cdot \left(\Br_{\mathrm{av}} - \Br_s\right),
\end{dmath}

with a difference of
%
\begin{dmath}\label{eqn:lecture11VectorDifference:170}
R_1 - R_2 \sim
-\frac{\Delta \Br}{r_{\mathrm{av}}} \cdot \left(\Br_{\mathrm{av}} - \Br_s\right).
\end{dmath}

} % makeanswer
