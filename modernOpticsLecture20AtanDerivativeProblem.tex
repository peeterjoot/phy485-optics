\makeproblem{arctan derivative.}{pr:modernOpticsLecture20AtanDerivativeProblem:1}{
Calculate

\begin{dmath}\label{eqn:modernOpticsLecture20AtanDerivativeProblem:650}
\ddx{}\Atan(x).
\end{dmath}
} % makeproblem

\makeanswer{pr:modernOpticsLecture20AtanDerivativeProblem:1}{

with

\begin{dmath}\label{eqn:modernOpticsLecture20AtanDerivativeProblem:670}
f(x) = \Atan(x),
\end{dmath}

We have

\begin{dmath}\label{eqn:modernOpticsLecture20AtanDerivativeProblem:690}
\tan f = x.
\end{dmath}

Taking derivatives

\begin{dmath}\label{eqn:modernOpticsLecture20AtanDerivativeProblem:710}
\frac{f'}{\cos^2(f)} = 1,
\end{dmath}

or

\begin{dmath}\label{eqn:modernOpticsLecture20AtanDerivativeProblem:730}
f' = \cos^2(f).
\end{dmath}

We've also got

\begin{dmath}\label{eqn:modernOpticsLecture20AtanDerivativeProblem:750}
x^2
=
\tan^2 f
= \frac{1 - \cos^2 f}{\cos^2 f}
\end{dmath}

or

\begin{dmath}\label{eqn:modernOpticsLecture20AtanDerivativeProblem:770}
f' = \cos^2 f = \inv{1 + x^2}
\end{dmath}

Our \(\phi(z)\) derivative then follows

\begin{dmath}\label{eqn:modernOpticsLecture20AtanDerivativeProblem:790}
\ddz{} \Atan(z/z_0)
=
\evalbar{\frac{d}{du} \Atan(u)}{u = z/z_0} \frac{d (z/z_0)}{dz}
=
\inv{1 + (z/z_0)^2} \inv{z_0}
= \frac{z_0}{z_0^2 + z^2}
\end{dmath}

} % makeanswer

\makeproblem{Derive expression for \(1/q\).}{pr:modernOpticsLecture20AtanDerivativeProblem:2}{
We have an expression for \(1/q\) in terms of the radius of curvature and waist functions in the handouts.  Derive this.
} % makeproblem

\makeanswer{pr:modernOpticsLecture20AtanDerivativeProblem:2}{

\begin{dmath}\label{eqn:modernOpticsLecture20AtanDerivativeProblem:810}
\inv{q}
=
\inv{z - i z_0}
=
\frac{z + i z_0}{z^2 + z_0^2}
=
\inv{R} +
\frac{i}{z_0( 1 + z^2/z_0^2)}
=
\inv{R} +
\frac{i w_0^2}{z_0 w^2(z)}
=
\inv{R} +
\frac{i \lambda z_0}{z_0 \pi w^2(z)}
\end{dmath}

or
\boxedEquation{eqn:modernOpticsLecture20AtanDerivativeProblem:830}{
\inv{q}
=
\inv{R} +
\frac{i \lambda}{\pi w^2(z)}.
}

} % makeanswer
